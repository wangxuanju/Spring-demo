# 注意的一些点
1.属性的必须是私有的

2.属性的数据类型一定正确的书写
int类型在实体类中是Integer
datetime类型是Timestamp(导入java.sql.Timestamp)

3.getter和setter方法尽量自动生成
```java
import java.sql.Timestamp;

public class Adamin {
    private Integer adamin_id;//属性的必须是私有的；
    private String adamin_code;
    private String password;
    private String name;
    private String telephone;
    private String eamil;
    private Timestamp enrolldate;
    
    //getter和setter方法尽量自动生成
    public Integer getAdamin_id() {
        return adamin_id;
    }

    public void setAdamin_id(Integer adamin_id) {
        this.adamin_id = adamin_id;
    }

    public String getAdamin_code() {
        return adamin_code;
    }

    public void setAdamin_code(String adamin_code) {
        this.adamin_code = adamin_code;
    }

    public String getPassword() {
        return password;
    }

    public void setPassword(String password) {
        this.password = password;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public String getTelephone() {
        return telephone;
    }

    public void setTelephone(String telephone) {
        this.telephone = telephone;
    }

    public String getEamil() {
        return eamil;
    }

    public void setEamil(String eamil) {
        this.eamil = eamil;
    }

    public Timestamp getEnrolldate() {
        return enrolldate;
    }

    public void setEnrolldate(Timestamp enrolldate) {
        this.enrolldate = enrolldate;
    }
}

```
